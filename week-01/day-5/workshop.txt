int main()
{

    printf("Irjon be ket szamot amik kozott vizsgalni szeretne a paratlan szamokat!\n");
    printf("A kezdo ertek legyen: \n");
    scanf("%d", &i);
    printf("A vegertek legyen: \n");
    scanf("%d", &j);

    printf(" \n");

    for(l = i; l <= j; l++) {

            if(l%2!=0) {
                    printf("%d, ", l);
                    m++;
            }
    }
    printf("\nThe number of odd numbers is: %d", m);

    return 0;
}
